Overview

This script recalculates the HWISE reduced score (excluding the angry and shame items) and fits the eight pre-specified regression models described in the analysis plan. Results are consolidated into a single publication-ready table.

Panel Outcome Model type Models
A Perceived Stress Score (PSS, 0–56) Linear regression m_a1 – m_a4
B Water-related emotion (positive / negative) Logistic regression m_b1 – m_b4

Covariates in all models: Age (D_AGE), SES score (SES_TOTAL), Household size (D_HH_SIZE). Panel B additionally adjusts for PSS (PSS_TOTAL).

HWISE_cat was defined using the cutoffs proposed by Young et al. (2021) for the full 12-item HWISE scale, which are based on the original validation sample from 8 countries. However, since we are using a reduced 10-item version of HWISE, these cutoffs may not be directly applicable. For the purposes of this analysis, we will use the same cutoffs as a starting point, but we will also explore the distribution of the reduced HWISE scores in our sample to see if they align with these categories or if adjustments are needed.

Recalculate HWISE Reduced Score

The HWISE reduced score sums the 10 items that remain after removing the emotion-laden items HW_ANGRY and HW_SHAME. A participant’s reduced score is set to NA if any of the 10 contributing items is missing.

# Items included in the reduced score (original 12 minus angry and shame)
hwise_items_reduced <- c(
  "HW_WORRY", "HW_INTERR", "HW_CLOTHES", "HW_PLANS",
  "HW_FOOD",  "HW_HANDS",  "HW_BODY",    "HW_DRINK",
  "HW_SLEEP", "HW_NONE"
)

hwise_items_full <- c(hwise_items_reduced, "HW_ANGRY", "HW_SHAME")

df <- df_raw %>%
  mutate(
    # Clean all HWISE items (reduced + emotion items)
    across(all_of(hwise_items_full), clean_num),

    # Reduced HWISE score: row sum, NA if any item is missing
    HWISE_REDUCED = rowSums(
      across(all_of(hwise_items_reduced)),
      na.rm = FALSE   # propagate NA if any item missing
    ),

    # Full HWISE score (all 12 items)
    HWISE_FULL = rowSums(
      across(all_of(hwise_items_full)),
      na.rm = FALSE
    ),

    # ── Outcomes ───────────────────────────────────────────
    PSS_TOTAL = clean_num(PSS_TOTAL),

    EMO_NEG = {
      raw <- clean_num(MX26_EM_HHW_CAT)
      case_when(
        raw == 0 ~ 0L,   # positive emotion
        raw == 1 ~ 1L,   # negative emotion
        TRUE     ~ NA_integer_
      )
    },

    # ── Core covariates ───────────────────────────────────────────
    D_AGE     = clean_num(D_AGE),
    SES_TOTAL = clean_num(SES_SC_Total),        # composite SES score
    D_HH_SIZE = clean_num(D_HH_SIZE),

    # ── Water-insecure neighbourhood (binary) ──────────────────────
    # W_WS_LOC: "WI" = water-insecure, "WS" = water-secure
    WI_NEIGHBORHOOD = factor(
      case_when(
        W_WS_LOC == "WI" ~ "Water-insecure",
        W_WS_LOC == "WS" ~ "Water-secure",
        TRUE             ~ NA_character_
      ),
      levels = c("Water-secure", "Water-insecure")  #water-secure = reference
    ),

    # ── Season ──────────────────────────────────────────────────────────────────
    # 0 = rainy season (reference), 1 = dry season
    SEASON = factor(
      SEASON,
      levels = c(0, 1),
      labels = c("Rainy", "Dry")
    )
  )

Polychoric PCA

Standard PCA assumes continuous, normally distributed variables. The HWISE items are ordinal (0–3 Likert scale), so PCA on the polychoric correlation matrix is more appropriate. Polychoric correlations estimate the latent Pearson correlation between pairs of assumed-continuous underlying variables that are observed as ordered categories.

swap_to_edu <- function(mod) {
  f_old  <- formula(mod)

  } else {

extract_comparison <- function(mods_edu, mods_ses, model_names, is_logistic = FALSE) {
  bind_rows(
    lapply(seq_along(mods_edu), function(i) {
      get_coefs <- function(mod, covariate_label) {
        tidy(mod, conf.int = TRUE, exponentiate = is_logistic) %>%
          filter(term %in% PRIMARY_TERMS) %>%
          mutate(
            model      = model_names[i],
            covariate  = covariate_label,
            term_label = recode(term, !!!TERM_LABELS, .default = term)
          ) %>%
          select(model, covariate, term_label, estimate, conf.low, conf.high, p.value)
      }
      bind_rows(
        get_coefs(mods_edu[[i]], "SES composite"),
        get_coefs(mods_ses[[i]], "EDU score")
      )
    })
  )
}

fmt_est <- function(est, lo, hi, p, digits = 3) {

          ifelse(p < 0.001, "***", ifelse(p < 0.01, "**",
                 ifelse(p < 0.05, "*", ifelse(p < 0.1, "†", "")))))
