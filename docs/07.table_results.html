Overview

This script recalculates the HWISE reduced score (excluding the angry and shame items) and fits the eight pre-specified regression models described in the analysis plan. Results are consolidated into a single publication-ready table.

Panel Outcome Model type Models
A Perceived Stress Score (PSS, 0–56) Linear regression m_a1 – m_a4
B Water-related emotion (positive / negative) Logistic regression m_b1 – m_b4

Covariates in all models: Age (D_AGE), SES score (SES_SC_Total), Household size (D_HH_SIZE). Panel B additionally adjusts for PSS (PSS_TOTAL).

MISSING_CODES <- c(999, 888, 777, 666, 555, 444, 333)

clean_num <- function(x) {
  x_num <- suppressWarnings(as.numeric(as.character(x)))
  ifelse(x_num %in% MISSING_CODES, NA_real_, x_num)
}

Recalculate HWISE Reduced Score

The HWISE reduced score sums the 10 items that remain after removing the emotion-laden items HW_ANGRY and HW_SHAME. A participant’s reduced score is set to NA if any of the 10 contributing items is missing.

# Items included in the reduced score (original 12 minus angry and shame)
hwise_items_reduced <- c(
  "HW_WORRY", "HW_INTERR", "HW_CLOTHES", "HW_PLANS",
  "HW_FOOD",  "HW_HANDS",  "HW_BODY",    "HW_DRINK",
  "HW_SLEEP", "HW_NONE"
)

df <- df_raw %>%
  mutate(
    # Clean all HWISE items
    across(all_of(hwise_items_reduced), clean_num),

    # Reduced HWISE score: row sum, NA if any item is missing
    HWISE_REDUCED = rowSums(
      across(all_of(hwise_items_reduced)),
      na.rm = FALSE   # propagate NA if any item missing
    ),

    # ── Outcomes ───────────────────────────────────────────
    PSS_TOTAL = clean_num(PSS_TOTAL),

    EMO_NEG = {
      raw <- clean_num(MX26_EM_HHW_CAT)
      case_when(
        raw == 0 ~ 0L,   # positive emotion
        raw == 1 ~ 1L,   # negative emotion
        TRUE     ~ NA_integer_
      )
    },

    # ── Core covariates ───────────────────────────────────────────
    D_AGE        = clean_num(D_AGE),
    SES_SC_Total = clean_num(SES_SC_Total),
    D_HH_SIZE    = clean_num(D_HH_SIZE),

    # ── Water-insecure neighbourhood (binary) ──────────────────────
    # W_WS_LOC: "WI" = water-insecure, "WS" = water-secure
    WI_NEIGHBORHOOD = factor(
      case_when(
        W_WS_LOC == "WI" ~ "Water-insecure",
        W_WS_LOC == "WS" ~ "Water-secure",
        TRUE             ~ NA_character_
      ),
      levels = c("Water-secure", "Water-insecure")  #water-secure = reference
    ),

    # ── Season ──────────────────────────────────────────────────────────────────
    # 0 = dry season (reference), 1 = rainy season
    SEASON = factor(
      SEASON,
      levels = c(0, 1),
      labels = c("Dry", "Rainy")
    )
  )

Model Specifications

# ── Panel A: PSS as outcome (linear regression) ─────────────────────────────
# Covariates: Age + SES + Household size (in every model)

m_a1 <- lm(
  PSS_TOTAL ~ HWISE_REDUCED + D_AGE + SES_SC_Total + D_HH_SIZE,
  data = df, na.action = na.omit
)

m_a2 <- lm(
  PSS_TOTAL ~ HWISE_REDUCED * SEASON + D_AGE + SES_SC_Total + D_HH_SIZE,
  data = df, na.action = na.omit
)

m_a3 <- lm(
  PSS_TOTAL ~ WI_NEIGHBORHOOD + D_AGE + SES_SC_Total + D_HH_SIZE,
  data = df, na.action = na.omit
)

m_a4 <- lm(
  PSS_TOTAL ~ WI_NEIGHBORHOOD * SEASON + D_AGE + SES_SC_Total + D_HH_SIZE,
  data = df, na.action = na.omit
)

# ── Panel B: Emotion type as outcome (logistic regression) ───────────────────
# Covariates: PSS + Age + SES + Household size (in every model)

m_b1 <- glm(
  EMO_NEG ~ HWISE_REDUCED + PSS_TOTAL + D_AGE + SES_SC_Total + D_HH_SIZE,
  data = df, family = binomial, na.action = na.omit
)

m_b2 <- glm(
  EMO_NEG ~ HWISE_REDUCED * SEASON + PSS_TOTAL + D_AGE + SES_SC_Total + D_HH_SIZE,
  data = df, family = binomial, na.action = na.omit
)

m_b3 <- glm(
  EMO_NEG ~ WI_NEIGHBORHOOD + PSS_TOTAL + D_AGE + SES_SC_Total + D_HH_SIZE,
  data = df, family = binomial, na.action = na.omit
)

m_b4 <- glm(
  EMO_NEG ~ WI_NEIGHBORHOOD * SEASON + PSS_TOTAL + D_AGE + SES_SC_Total + D_HH_SIZE,
  data = df, family = binomial, na.action = na.omit
)

Season-Stratified Estimates

Interaction models (m_a2, m_a4, m_b2, m_b4) are complemented by season- stratified re-runs to facilitate interpretation.

seasons <- c("Dry", "Rainy")

# ── Panel A: stratified ───────────────────────────────────────────────────────
strat_a <- map_dfr(seasons, function(s) {
  df_s <- filter(df, SEASON == s)
  list(
    hwise  = lm(PSS_TOTAL ~ HWISE_REDUCED + D_AGE + SES_SC_Total + D_HH_SIZE,
                data = df_s, na.action = na.omit),
    wi_nbh = lm(PSS_TOTAL ~ WI_NEIGHBORHOOD + D_AGE + SES_SC_Total + D_HH_SIZE,
                data = df_s, na.action = na.omit)
  ) %>%
    map_dfr(~ tidy(.x, conf.int = TRUE), .id = "predictor") %>%
    filter(!term %in% c("(Intercept)", "D_AGE", "SES_SC_Total", "D_HH_SIZE")) %>%
    mutate(season = s)
}) %>%
  mutate(
    ci_fmt = sprintf("%.2f [%.2f, %.2f]", estimate, conf.low, conf.high),
    p_fmt  = case_when(p.value < 0.001 ~ "<0.001",
                       TRUE ~ sprintf("%.3f", p.value))
  ) %>%
  select(season, predictor, term, n = statistic, ci_fmt, p_fmt)
