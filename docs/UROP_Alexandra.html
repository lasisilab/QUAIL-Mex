hw_vars <- c("HW_WORRY", "HW_INTERR", "HW_CLOTHES", "HW_PLANS", "HW_FOOD",
             "HW_HANDS", "HW_BODY", "HW_DRINK", "HW_ANGRY", "HW_SLEEP",

# Recode mother status as factor
d$D_CHLD_CAT <- factor(ifelse(d$D_CHLD >= 1, "Mother", "Non-mother"), levels = c("Non-mother", "Mother"))

# Convert HW variables to factors
d[hw_vars] <- lapply(d[hw_vars], function(x) factor(x, levels = c(0, 1)))

# Initialize test results list
results <- data.frame()

# Loop over each HW variable to test and collect proportions
for (var in hw_vars) {
  tbl <- table(d$D_CHLD_CAT, d[[var]])
  
  # Run chi-square or fisher test based on counts
  if (any(tbl < 5)) {
    test <- fisher.test(tbl)
    method <- "Fisher"
  } else {
    test <- chisq.test(tbl)
    method <- "Chi-square"
  }
  
  # Get proportions for each group
  prop_table <- prop.table(tbl, margin = 1)[, "1"]
  
  # Store results
  results <- rbind(results, data.frame(
    Variable = var,
    Non_mother = round(prop_table["Non-mother"], 3),
    Mother = round(prop_table["Mother"], 3),
    p_value = round(test$p.value, 4),
    test_type = method
  ))
}
