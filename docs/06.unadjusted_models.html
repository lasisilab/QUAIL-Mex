Overview

This script fits unadjusted (univariable) regression models for all candidate predictors against two outcomes. Each predictor is tested in a separate model with no adjustment for covariates. Results serve as a reference for effect direction and magnitude before covariate adjustment in files 07–10.

Panel Outcome Model type
A Perceived Stress Score (PSS_TOTAL, 0–56) Linear regression (β)
B Water-related emotion valence (EMO_NEG) Logistic regression (OR)

Predictors are organised into six groups: Demographics, Socioeconomic, Health, HWISE (composite and subscale measures), HWISE items (individual items), and Water supply characteristics.

  • Summary tables include all predictor groups; HWISE categorical (HWISE_Cat) and HWISE binary high/low (HWISE_Hi) are shown in tables but excluded from forest plots due to sparse cell counts.
  • Multiple testing: Significance thresholds and row highlighting in all tables and plots are based on Benjamini–Hochberg (BH) adjusted q-values, applied across all predictors within each outcome.

# Build a display label that appends level & ref for categorical terms
make_display_label <- function(predictor, term, label) {
  if (predictor == term) return(label)          # continuous
  level <- gsub(predictor, "", term, fixed = TRUE)
  ref   <- ref_levels[predictor]
  if (!is.na(ref)) {
    paste0(label, ": ", level, " (ref: ", ref, ")")
  } else {
    paste0(label, ": ", level)
  }
}

# Helper: drop sparse levels and run unadjusted linear regression
run_lm_uv <- function(outcome, predictors, data, labels) {
  map_dfr(predictors, function(v) {
    dat <- data[, c(outcome, v)]
    for (drop_lvl in c("Variable", "Other")) {
      if (v %in% c("W_Pressure", "W_Frequency", "W_Continuity") &&
          drop_lvl %in% levels(dat[[v]])) {
        dat <- dat[is.na(dat[[v]]) | dat[[v]] != drop_lvl, ]
        dat[[v]] <- droplevels(dat[[v]])
      }
    }
    dat <- dat[complete.cases(dat), ]
    fit <- tryCatch(
      lm(as.formula(paste(outcome, "~ .")), data = dat),
      error = function(e) NULL
    )
    if (is.null(fit)) return(NULL)
    tidy(fit, conf.int = TRUE) %>%
      filter(term != "(Intercept)") %>%
      mutate(predictor = v, label = labels[v], AIC = AIC(fit), n = nrow(dat))
  })
}

# Helper: drop sparse levels and run unadjusted logistic regression
run_glm_uv <- function(outcome, predictors, data, labels) {
  map_dfr(predictors, function(v) {
    dat <- data[, c(outcome, v)]
    for (drop_lvl in c("Variable", "Other")) {
      if (v %in% c("W_Pressure", "W_Frequency", "W_Continuity") &&
          drop_lvl %in% levels(dat[[v]])) {
        dat <- dat[is.na(dat[[v]]) | dat[[v]] != drop_lvl, ]
        dat[[v]] <- droplevels(dat[[v]])
      }
    }
    dat <- dat[complete.cases(dat), ]
    fit <- tryCatch(
      glm(as.formula(paste(outcome, "~ .")), data = dat, family = binomial),
      error = function(e) NULL
    )
    if (is.null(fit)) return(NULL)
    tidy(fit, conf.int = TRUE, exponentiate = TRUE) %>%
      filter(term != "(Intercept)") %>%
      mutate(predictor = v, label = labels[v], AIC = AIC(fit), n = nrow(dat))
  })
}

make_uv_flextable <- function(tbl, sig_rows, title_str, footnote_str,
                               hi_color = "#fff3cd") {
  tbl %>%
    select(-p.value) %>%
    flextable() %>%
    set_header_labels(
      Variable       = "Predictor",
      N              = "n",
      Sig            = "Sig.",
      `p-value`      = "p-value",
      AIC            = "AIC"
    ) %>%
    # Title row above header
    add_header_lines(title_str) %>%
    # Fonts
    font(fontname = "Calibri", part = "all") %>%
    fontsize(size = 10, part = "body") %>%
    fontsize(size = 10, part = "header") %>%
    fontsize(size = 9,  part = "footer") %>%
    # Header styling
    bold(part = "header") %>%
    color(part = "header", color = "white") %>%
    bg(part = "header", bg = "#1A3A5C") %>%
    # Title row styling (first header row = title)
    fontsize(i = 1, part = "header", size = 11) %>%
    bold(i = 1, part = "header") %>%
    # Alternating row shading
    bg(i = seq(2, nrow(tbl %>% select(-p.value)), by = 2),
       bg = "#F9F9F9", part = "body") %>%
    # Significant row highlight
    bold(i = sig_rows) %>%
    bg(i = sig_rows, bg = hi_color) %>%
    # Borders
    border_remove() %>%
    hline_top(part = "header",
              border = fp_border(color = "#1A3A5C", width = 1.5)) %>%
    hline_bottom(part = "header",
                 border = fp_border(color = "#1A3A5C", width = 1.5)) %>%
    hline_bottom(part = "body",
                 border = fp_border(color = "#1A3A5C", width = 1.5)) %>%
    hline(border = fp_border(color = "#bdc3c7", width = 0.5), part = "body") %>%
    # Alignment
    align(j = 1,  align = "left",   part = "all") %>%
    align(j = -1, align = "center", part = "all") %>%
    # Footer
    add_footer_lines(footnote_str) %>%
    italic(part = "footer") %>%
    color(part = "footer", color = "#555555") %>%
    # Layout
    padding(padding.top = 2, padding.bottom = 2, part = "all") %>%
    set_table_properties(layout = "autofit", width = 1) %>%
    fix_border_issues()
}

insert_group_headers_uv <- function(df, group_col = "Group") {
  groups <- unique(df[[group_col]])
  purrr::map_dfr(groups, function(g) {
    rows       <- df[df[[group_col]] == g, ]
    header_row <- rows[1, ]
    for (col in names(header_row)) header_row[[col]] <- NA
    header_row[[group_col]]   <- g
    header_row[["Predictor"]] <- g
    bind_rows(header_row, rows)
  })
}
