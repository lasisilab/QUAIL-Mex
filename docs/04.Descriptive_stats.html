Introduction

This document produces descriptive statistics tables for the merged dataset (3.Merged_Q1-6_Q16_Q19_Q26_Q28.csv, produced by 03.Data_merge_Q1-6.Rmd). It includes:

  1. A comprehensive descriptive statistics table (all analytic variables)

  2. A manuscript table with sample characteristics stratified by season

  3. A missing data summary table

  4. HWISE and PSS descriptive statistics (overall and by season)

  5. Sample characteristics stratified by water insecurity level (HWISE categories)

  6. Objective water supply characteristics (source, continuity, frequency, hours of supply, pressure, and seasonality)

Visualizations for all of the above are in 05.Descriptive_plots.Rmd.

# Helper: convert sentinel codes to NA for descriptive stats
clean_for_table <- function(x) {
  x_num <- suppressWarnings(as.numeric(as.character(x)))
  ifelse(x_num %in% MISSING_CODES | is.na(x_num), NA, x_num)
}

# Helper: flag sentinel codes in free-text / character fields
SENTINEL_STRINGS <- as.character(c(999, 888, 777, 666, 555, 444, 333))
is_sentinel_str <- function(x) {
  str_trim(as.character(x)) %in% SENTINEL_STRINGS
}

# Helper: build water supply variables from a raw df
make_water_vars <- function(df) {
  df %>%
    mutate(
      W_Continuity = case_when(
        is_sentinel_str(MX3_CONT_INT)                             ~ NA_character_,
        str_detect(tolower(MX3_CONT_INT), "^contin")              ~ "Continuous",
        str_detect(tolower(MX3_CONT_INT), "^intermi|cortan|deja") ~ "Intermittent",
        MX3_CONT_INT %in% c("", "N/A") | is.na(MX3_CONT_INT)    ~ NA_character_,
        TRUE                                                       ~ "Other"
      ),
      W_Continuity = factor(W_Continuity,
                            levels = c("Continuous", "Intermittent", "Other")),
      W_Freq_raw  = coalesce(MX4_WFREQ, MX4_W_FREQ),
      W_Frequency = case_when(
        is_sentinel_str(W_Freq_raw)                               ~ NA_character_,
        str_detect(tolower(W_Freq_raw), "^diari|every day")       ~ "Daily",
        str_detect(tolower(W_Freq_raw), "1 a 3|1-3")              ~ "1–3 days/week",
        str_detect(tolower(W_Freq_raw), "4 a 6|4-6")              ~ "4–6 days/week",
        W_Freq_raw %in% c("", "N/A") | is.na(W_Freq_raw)        ~ NA_character_,
        TRUE                                                       ~ NA_character_
      ),
      W_Frequency = factor(W_Frequency,
                           levels = c("Daily", "4–6 days/week", "1–3 days/week")),
      W_Days_week_num = suppressWarnings(
        ifelse(is_sentinel_str(MX5_WFREQ_NDAYS), NA,
               as.numeric(str_extract(MX5_WFREQ_NDAYS, "^[0-9]+")))
      ),
      W_Days_week_num = ifelse(W_Days_week_num %in% MISSING_CODES | W_Days_week_num > 7,
                               NA, W_Days_week_num),
      W_Days_week = factor(
        case_when(
          is.na(W_Days_week_num) ~ NA_character_,
          W_Days_week_num <= 2   ~ "1–2 days/week",
          W_Days_week_num <= 5   ~ "3–5 days/week",
          TRUE                   ~ "6–7 days/week"
        ),
        levels = c("1–2 days/week", "3–5 days/week", "6–7 days/week")
      ),
      W_Hrs_week  = suppressWarnings(
        ifelse(is_sentinel_str(MX5_WFREQ_HRS_PER_WEEK), NA,
               as.numeric(str_extract(MX5_WFREQ_HRS_PER_WEEK, "^[0-9\\.]+")))
      ),
      W_Hrs_week  = ifelse(W_Hrs_week %in% MISSING_CODES | W_Hrs_week > 168,
                           NA, W_Hrs_week),
      W_Pressure  = case_when(
        is_sentinel_str(MX6_WPRESS)                                  ~ NA_character_,
        str_detect(tolower(MX6_WPRESS), "^s[ií]|^yes|^buena|^bien") ~ "Yes",
        str_detect(tolower(MX6_WPRESS), "^no|^baja|^poca")          ~ "No/Low",
        MX6_WPRESS %in% c("", "N/A") | is.na(MX6_WPRESS)          ~ NA_character_,
        TRUE                                                          ~ "Variable"
      ),
      W_Pressure  = factor(W_Pressure, levels = c("Yes", "No/Low", "Variable"))
    )
}

miss_pvals_tbl <- sapply(miss_cols, function(v) {
  tbl <- table(is.na(df_miss[[v]]), df_miss$SEASON)
  if (any(dim(tbl) < 2) || nrow(tbl) < 2) return(NA_real_)
  tryCatch(
    if (any(tbl < 5)) fisher.test(tbl)$p.value
    else              chisq.test(tbl, correct = FALSE)$p.value,
    error = function(e) NA_real_
  )
})

9. Internal consistency (Cronbach’s α)

Cronbach’s α measures how well the items in a scale correlate with each other — i.e., whether they are all capturing the same underlying construct. Values ≥ 0.7 are generally considered acceptable. α is computed here on complete cases only (rows with any NA or sentinel code in the relevant items are excluded).

clean_items <- function(data, items) {
  data %>%
    select(all_of(items)) %>%
    mutate(across(everything(), ~ if_else(. %in% MISSING_CODES, NA_real_,
                                          as.numeric(.)))) %>%
    filter(if_all(everything(), ~ !is.na(.)))
}

# Helper: convert a gtsummary table to a formatted flextable ready for Word
fmt_docx <- function(tbl) {
  ft <- tbl %>% as_flex_table()
  n_body <- nrow(ft$body$dataset)
  ft %>%
    font(fontname = "Calibri", part = "all") %>%
    fontsize(size = 10, part = "body") %>%
    fontsize(size = 10, part = "header") %>%
    fontsize(size = 9,  part = "footer") %>%
    # Header: dark navy background, white bold text
    bg(bg = "#1A3A5C", part = "header") %>%
    color(color = "white", part = "header") %>%
    bold(part = "header") %>%
    # Alternating row shading
    bg(i = seq(2, n_body, by = 2), bg = "#F9F9F9", part = "body") %>%
    # Horizontal-only borders
    border_remove() %>%
    hline_top(part = "header",
              border = fp_border(color = "#1A3A5C", width = 1.5)) %>%
    hline_bottom(part = "header",
                 border = fp_border(color = "#1A3A5C", width = 1.5)) %>%
    hline_bottom(part = "body",
                 border = fp_border(color = "#1A3A5C", width = 1.5)) %>%
    hline(border = fp_border(color = "#bdc3c7", width = 0.5), part = "body") %>%
    # Align: label column left, stats/p-value columns centre
    align(j = 1, align = "left",   part = "all") %>%
    align(j = -1, align = "center", part = "all") %>%
    # Compact row height and padding
    padding(padding.top = 2, padding.bottom = 2, part = "all") %>%
    # Footer: italic, grey
    italic(part = "footer") %>%
    color(color = "#555555", part = "footer") %>%
    set_table_properties(layout = "autofit", width = 1) %>%
    fix_border_issues()
}
